2.3 수수료 계산 로직

1) 엔터 베이직

-- 신규 고객
월 수수료 = SUM(전체 거래 매출) × 2%

WHERE 
  거래일자 BETWEEN 당월_1일 AND 당월_말일
  AND 취소/환불 건 제외

-- 무료 제공 조건 체크
IF (월 수수료 >= 5,000,000 AND 광고주) OR 기존_모멘텀_광고주 THEN
  월 수수료 = 0원
END

-- 업셀 기준 체크
IF 월 수수료 >= 5,000,000 THEN
  → 엔터프라이즈 제안 알림
END
계산 예시
고객사 A (일반 고객)
- 월 매출: 1억원
- 수수료: 1억 × 2% = 200만원
→ 200만원 청구

고객사 B (광고주)
- 월 매출: 3억원
- 수수료: 3억 × 2% = 600만원 (>500만원)
- 광고주 여부: YES
→ 0원 (무료 제공)
→ 엔터프라이즈 제안 발송

고객사 C (기존 모멘텀 광고주)
- 월 매출: 5천만원
- 수수료: 100만원
→ 0원 (무료 제공)

2) 엔터프라이즈

정률제:

일별 수수료 = SUM(전체 거래 매출) × fee_rate

-- 최소/최대 수수료 적용
IF min_fee IS NOT NULL AND 일별수수료 < min_fee THEN 일별수수료 = min_fee
IF max_fee IS NOT NULL AND 일별수수료 > max_fee THEN 일별수수료 = max_fee

정액제:

월 정액 = fixed_amount × (이용일수 / 해당월_총일수)

CREATE TABLE enterprise_basic_policy (
    policy_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    customer_id VARCHAR(50) NOT NULL,
    
    -- 고객 유형
    customer_type VARCHAR(20) NOT NULL, -- 'NORMAL', 'ADVERTISER', 'MOMENTUM_ADVERTISER'
    
    -- 무료 제공 여부
    is_free BOOLEAN DEFAULT FALSE,
    free_reason VARCHAR(100), -- '월 500만원 이상 광고주', '기존 모멘텀 광고주'
    
    -- 업셀 관리
    monthly_revenue BIGINT, -- 최근 월 매출
    monthly_fee BIGINT,     -- 최근 월 수수료
    upsell_eligible BOOLEAN DEFAULT FALSE, -- 엔터프라이즈 제안 대상
    
    -- 적용 기간
    start_date DATE NOT NULL,
    end_date DATE,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    is_active BOOLEAN DEFAULT TRUE,
    
    INDEX idx_customer (customer_id),
    INDEX idx_upsell (upsell_eligible, monthly_fee)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE enterprise_billing_policy (
    policy_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    customer_id VARCHAR(50) NOT NULL,
    
    fee_type VARCHAR(20) NOT NULL,  -- 'PERCENTAGE', 'FIXED', 'FREE'
    
    -- 정률제
    fee_rate DECIMAL(5,2),
    min_fee BIGINT,
    max_fee BIGINT,
    
    -- 정액제
    fixed_amount BIGINT,
    
    -- 무료
    free_reason VARCHAR(100),
    
    start_date DATE NOT NULL,
    end_date DATE,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    is_active BOOLEAN DEFAULT TRUE,
    
    INDEX idx_customer (customer_id),
    INDEX idx_fee_type (fee_type)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
